#include <iostream>

using namespace std;

// Function to check if a year is a leap year
bool isLeapYear(int year) {
    return (year % 400 == 0) || (year % 4 == 0 && year % 100 != 0);
}

// Function to get the century value
int getCenturyValue(int year) {
    int century = year / 100;
    int remainder = century % 4;
    return (3 - remainder) * 2;
}

// Function to get the year value
int getYearValue(int year) {
    int last_two_digits = year % 100;
    return last_two_digits + (last_two_digits / 4);
}

// Function to get the month value
int getMonthValue(int month, int year) {
    int month_values[] = {0, 3, 3, 6, 1, 4, 6, 2, 5, 0, 3, 5}; // Values for Jan to Dec
    if (month == 1 && isLeapYear(year)) {  // January in a leap year
        return 6;
    } else if (month == 2 && isLeapYear(year)) {  // February in a leap year
        return 2;
    }
    return month_values[month - 1];
}

// Main function to compute the day of the week
int main() {
    int month, day, year;
    cout << "Enter date (mm dd yyyy): ";
    cin >> month >> day >> year;

    // Calculate the components
    int century_value = getCenturyValue(year);
    int year_value = getYearValue(year);
    int month_value = getMonthValue(month, year);

    // Calculate the total value
    int total = (day + month_value + year_value + century_value) % 7;

    // Determine the day of the week
    string days_of_week[] = {"Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"};
    cout << "Day of the week is: " << days_of_week[total] << endl;

    return 0;
}

#include <iostream>

using namespace std;

// Function to compute the number of coins and update the amount left
void computeCoin(int coinValue, int& number, int& amountLeft) {
    number = amountLeft / coinValue;       // Maximum number of coins
    amountLeft = amountLeft % coinValue;   // Update the remaining amount
}

int main() {
    char choice = 'Y';
    while (choice == 'Y' || choice == 'y') {
        int amountLeft, quarters, dimes, pennies;

        // Input amount of change from user
        cout << "Enter an amount of change (1 to 99 cents): ";
        cin >> amountLeft;

        if (amountLeft < 1 || amountLeft > 99) {
            cout << "Please enter a valid amount between 1 and 99 cents." << endl;
            continue;
        }

        // Compute number of quarters
        computeCoin(25, quarters, amountLeft);

        // Compute number of dimes
        computeCoin(10, dimes, amountLeft);

        // Compute number of pennies
        computeCoin(1, pennies, amountLeft);

        // Output the result
        cout << "Quarters: " << quarters << endl;
        cout << "Dimes: " << dimes << endl;
        cout << "Pennies: " << pennies << endl;

        // Ask if user wants to calculate another change
        cout << "Do you want to calculate another amount? (Y/N): ";
        cin >> choice;
    }

    return 0;
}

#include <iostream>

using namespace std;

// Function to convert pace from minutes and seconds to miles per hour
double convertToMPH(int minutes, int seconds) {
    double totalMinutes = minutes + seconds / 60.0;
    double milesPerHour = 60.0 / totalMinutes;  // Speed in miles per hour
    return milesPerHour;
}

// Overloaded function to convert kilometers per hour to miles per hour
double convertToMPH(double kilometersPerHour) {
    return kilometersPerHour / 1.61;  // Convert km/h to mph
}

int main() {
    int choice;

    cout << "Choose the type of conversion: " << endl;
    cout << "1. Convert from minutes and seconds per mile to MPH" << endl;
    cout << "2. Convert from kilometers per hour to MPH" << endl;
    cout << "Enter your choice (1 or 2): ";
    cin >> choice;

    if (choice == 1) {
        int minutes, seconds;
        cout << "Enter minutes: ";
        cin >> minutes;
        cout << "Enter seconds: ";
        cin >> seconds;

        // Call the first overloaded function
        double mph = convertToMPH(minutes, seconds);
        cout << "Speed in MPH: " << mph << endl;
    } else if (choice == 2) {
        double kilometersPerHour;
        cout << "Enter speed in kilometers per hour: ";
        cin >> kilometersPerHour;

        // Call the second overloaded function
        double mph = convertToMPH(kilometersPerHour);
        cout << "Speed in MPH: " << mph << endl;
    } else {
        cout << "Invalid choice!" << endl;
    }

    return 0;
}
